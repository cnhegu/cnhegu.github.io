<script>
    const progressBar = document.querySelector('.progress-bar');
    const postContent = document.querySelector('article'); // 假设您的文章主体在<article>标签内

    function updateProgressBar() {
        const scrollableHeight = postContent.offsetHeight - window.innerHeight;
        const scrollTop = window.scrollY;
        const progress = (scrollTop / scrollableHeight) * 100;

        if (progressBar) {
            progressBar.style.width = (progress > 100 ? 100 : progress) + '%';
        }
    }

    window.addEventListener('scroll', updateProgressBar);
    updateProgressBar(); // 初始加载时也更新一次
</script>
